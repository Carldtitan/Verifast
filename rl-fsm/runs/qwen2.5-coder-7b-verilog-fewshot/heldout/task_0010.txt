module fsm_arm (
    input  logic clk,
    input  logic rst,
    input  logic i0,
    input  logic i1,
    input  logic i2,
    output logic [1:0] o0
);

    typedef enum logic [2:0] { S0, S1, S2, S3, S4 } state_t;
    state_t state, next_state;

    // Combinational next-state logic (blocking assignments only).
    always_comb begin
        next_state = state;
        case (state)
            S0: begin
                o0 = 2'd1;
                if (i0) next_state = S2;
                else    next_state = S4;
            end
            S1: begin
                o0 = 2'd0;
                if (i2) next_state = S3;
                else    next_state = S0;
            end
            S2: begin
                o0 = 2'd2;
                if (i0) next_state = S4;
                else    next_state = S2;
            end
            S3: begin
                o0 = 2'd2;
                if (i1) next_state = S1;
                else    next_state = S3;
            end
            S4: begin
                o0 = 2'd2;
                if (i1) next_state = S1;
                else    next_state = S0;
            end
            default: next_state = state;
        endcase
    end

    // Combinational Moore output logic (function of state only).
    always_comb begin
        o0 = 2'd0;
        case (state)
            S0: o0 = 2'd1;
            S1: o0 = 2'd0;
            S2: o0 = 2'd2;
            S3: o0 = 2'd2;
            S4: o0 = 2'd2;
            default: o0 = 2'd0;
        endcase
    end

    // Sequential state register with synchronous active-high reset.
    always_ff @(posedge clk) begin
        if (rst) state <= S0;
        else     state <= next_state;
    end

endmodule
