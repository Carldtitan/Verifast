module fsm_idle (
    input  logic clk,
    input  logic rst,
    input  logic i0,
    input  logic i1,
    input  logic i2,
    output logic o0
);

    // State encoding
    typedef enum logic [1:0] {
        S0 = 2'b00,
        S1 = 2'b01,
        S2 = 2'b10,
        S3 = 2'b11
    } state_t;

    state_t current_state, next_state;

    // State register
    always_ff @(posedge clk) begin
        if (rst) begin
            current_state <= S0;
        end else begin
            current_state <= next_state;
        end
    end

    // Next state logic
    always_comb begin
        next_state = current_state;
        case (current_state)
            S0: begin
                if (i1)
                    next_state = S2;
                else
                    next_state = S1;
            end
            
            S1: begin
                if (i2)
                    next_state = S2;
                else
                    next_state = S0;
            end
            
            S2: begin
                if (i1)
                    next_state = S3;
                else
                    next_state = S0;
            end
            
            S3: begin
                next_state = S3;
            end
            
            default: next_state = S0;
        endcase
    end

    // Output logic (Moore machine - outputs depend only on current state)
    always_comb begin
        case (current_state)
            S0: o0 = 1'b0;
            S1: o0 = 1'b0;
            S2: o0 = 1'b1;
            S3: o0 = 1'b1;
            default: o0 = 1'b0;
        endcase
    end

endmodule
