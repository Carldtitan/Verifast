module fsm_arm (
    input  logic       clk,
    input  logic       rst,
    input  logic       i0,
    input  logic       i1,
    input  logic       i2,
    output logic [1:0] o0
);

    // State encoding
    typedef enum logic [2:0] {
        S0 = 3'b000,
        S1 = 3'b001,
        S2 = 3'b010,
        S3 = 3'b011,
        S4 = 3'b100
    } state_t;

    state_t current_state, next_state;

    // State register
    always_ff @(posedge clk) begin
        if (rst) begin
            current_state <= S0;
        end else begin
            current_state <= next_state;
        end
    end

    // Next state logic
    always_comb begin
        next_state = current_state; // Default assignment
        case (current_state)
            S0: begin
                if (i0)
                    next_state = S2;
                else
                    next_state = S4;
            end
            S1: begin
                if (i2)
                    next_state = S3;
                else
                    next_state = S0;
            end
            S2: begin
                if (i0)
                    next_state = S4;
                else
                    next_state = S2;
            end
            S3: begin
                if (i1)
                    next_state = S1;
                else
                    next_state = S3;
            end
            S4: begin
                next_state = S1;
            end
            default: next_state = S0;
        endcase
    end

    // Output logic (Moore machine - outputs depend only on current state)
    always_comb begin
        case (current_state)
            S0: o0 = 2'd1;
            S1: o0 = 2'd0;
            S2: o0 = 2'd2;
            S3: o0 = 2'd2;
            S4: o0 = 2'd2;
            default: o0 = 2'd0;
        endcase
    end

endmodule
