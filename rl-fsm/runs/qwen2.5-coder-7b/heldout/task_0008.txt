module fsm_lock (
    input wire clk,
    input wire rst,
    input wire i0,
    input wire i1,
    output reg o0,
    output reg o1
);

    // Define states
    localparam S0 = 2'b00;
    localparam S1 = 2'b01;

    // State register
    reg [1:0] state;

    // Next state logic
    always @(posedge clk or posedge rst) begin
        if (rst) begin
            state <= S0;
        end else begin
            case (state)
                S0: begin
                    o0 <= 1;
                    o1 <= 1;
                    if (i0) begin
                        state <= S0;
                    end else begin
                        state <= S1;
                    end
                end
                S1: begin
                    o0 <= 0;
                    o1 <= 0;
                    if (i0) begin
                        state <= S0;
                    end else if (i1) begin
                        state <= S1;
                    end else begin
                        state <= S0;
                    end
                end
                default: state <= S0;
            endcase
        end
    end

endmodule
