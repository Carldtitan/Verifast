module fsm_arm (
    input i0,
    output reg [1:0] o0
);

    // State definitions
    localparam S0 = 2'b00,
               S1 = 2'b01,
               S2 = 2'b10,
               S3 = 2'b11;

    // Current state register
    reg [1:0] state;

    // Next state logic
    always @(posedge clk or posedge rst) begin
        if (rst) begin
            state <= S0;
        end else begin
            case (state)
                S0: state <= i0 ? S2 : S0;
                S1: state <= i0 ? S3 : S1;
                S2: state <= i0 ? S3 : S2;
                S3: state <= i0 ? S0 : S3;
                default: state <= S0;
            endcase
        end
    end

    // Output logic
    always @(*) begin
        case (state)
            S0: o0 = 2'b01;
            S1: o0 = 2'b01;
            S2: o0 = 2'b10;
            S3: o0 = 2'b11;
            default: o0 = 2'b00;
        endcase
    end

endmodule
