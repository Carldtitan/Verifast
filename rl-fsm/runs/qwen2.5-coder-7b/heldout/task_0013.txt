module fsm_hold (
    input clk,
    input rst,
    input i0,
    input i1,
    output reg [1:0] o0,
    output reg o1
);

    // Define states
    typedef enum reg [1:0] {S0, S1} state_t;
    state_t state, next_state;

    // State transition logic
    always @(posedge clk or posedge rst) begin
        if (rst) begin
            state <= S0;
        end else begin
            state <= next_state;
        end
    end

    // Output logic
    always @(*) begin
        case (state)
            S0: begin
                o0 = 2;
                o1 = 0;
                if (i0) begin
                    next_state = S0;
                end else begin
                    next_state = S0;
                end
            end
            S1: begin
                o0 = 2;
                o1 = 1;
                if (i0) begin
                    next_state = S0;
                end else begin
                    next_state = S1;
                end
            end
            default: begin
                o0 = 2;
                o1 = 0;
                next_state = S0;
            end
        endcase
    end

endmodule
