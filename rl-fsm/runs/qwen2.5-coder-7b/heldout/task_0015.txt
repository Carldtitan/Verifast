module fsm_arm (
    input i0,
    input i1,
    output reg o0
);

    // Define states
    localparam S0 = 2'b00;
    localparam S1 = 2'b01;
    localparam S2 = 2'b10;
    localparam S3 = 2'b11;

    // State register
    reg [1:0] state;

    // Next state logic
    always @(posedge clk or posedge rst) begin
        if (rst) begin
            state <= S0;
        end else begin
            case (state)
                S0: begin
                    o0 <= 0;
                    if (i1) begin
                        state <= S2;
                    end else begin
                        state <= S3;
                    end
                end
                S1: begin
                    o0 <= 0;
                    if (i0) begin
                        state <= S1;
                    end
                end
                S2: begin
                    o0 <= 1;
                    if (i0) begin
                        state <= S3;
                    end else begin
                        state <= S0;
                    end
                end
                S3: begin
                    o0 <= 0;
                    if (i1) begin
                        state <= S3;
                    end else if (i0) begin
                        state <= S3;
                    end else begin
                        state <= S0;
                    end
                end
                default: state <= S0;
            endcase
        end
    end

endmodule
