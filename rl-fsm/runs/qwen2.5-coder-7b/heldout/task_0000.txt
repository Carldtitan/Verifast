module fsm_scan (
    input i0,
    input i1,
    input i2,
    output reg o0,
    output reg o1
);

    // State definitions
    localparam S0 = 2'b00;
    localparam S1 = 2'b01;

    // State register
    reg [1:0] state;

    // Next state logic
    reg [1:0] next_state;

    always @(posedge clk or posedge rst) begin
        if (rst) begin
            state <= S0;
        end else begin
            state <= next_state;
        end
    end

    // Output logic
    always @(*) begin
        case (state)
            S0: begin
                o0 = 0;
                o1 = 0;
            end
            S1: begin
                o0 = 0;
                o1 = 1;
            end
            default: begin
                o0 = 0;
                o1 = 0;
            end
        endcase
    end

    // Next state logic
    always @(*) begin
        case (state)
            S0: begin
                if (i0 || i1 || i2) begin
                    next_state = S1;
                end else begin
                    next_state = S0;
                end
            end
            S1: begin
                if (i0) begin
                    next_state = S1;
                end else begin
                    next_state = S0;
                end
            end
            default: begin
                next_state = S0;
            end
        endcase
    end

endmodule
